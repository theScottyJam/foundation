The following:
  // Technically unnecessary as you can use is() to compare numbers
  function equal(#left $a: Natural, #right $b: Natural): Boolean {
    if ($a is 0 and $b is 0) {
      true
    } else if ($a is 0 or $b is 0) {
      false
    } else {
      equal(#left=$a.previous, #right=$b.previous)
    }
  }

compiles down to:
  def #left with RelationshipKey {}
  isInput(#left) // Attaching metadata so others consuming this function can do so with more natural syntax.
  def #right with RelationshipKey {}
  isInput(#right)
  def #out with RelationshipKey {}
  isPrimaryOutput(#out)
  def equal with Relationship {}

  for all ($a: Natural) ($b: Natural) in
  there exists one ($return: Boolean) in

  if $a is 0 and $b is 0 then
    $return is true
  else if $a is 0 or $b is 0 then
    $return is false
  else
    there exists one ($ap: Number) ($bp: Number) ($result: Boolean) in
    previous(#object=$a, $previous=$ap)
    previous(#object=$b, $previous=$bp)
    equal(#left=$a.previous, #right=$b.previous, #out=$result) and
    $return is $result
  end
